 
SELECT C.C_NUM, C.C_NAME, C.CITY, C.RATING, O.O_NUM, O.AMT, O.ORDER_DATE
FROM CUSTOMERS C
JOIN ORDERS O ON C.C_NUM = O.C_NUM
WHERE O.AMT > 5000 AND YEAR(O.ORDER_DATE) = 2024;


SELECT *
FROM CUSTOMERS
WHERE C_NUM NOT IN (SELECT DISTINCT C_NUM FROM ORDERS);


SELECT C_NAME, CITY
FROM CUSTOMERS
WHERE RATING = (SELECT RATING FROM CUSTOMERS WHERE C_NAME = 'Hoffman');


SELECT S_NUM, SUM(AMT) AS TOTAL_AMOUNT
FROM ORDERS
GROUP BY S_NUM
HAVING SUM(AMT) > (SELECT MAX(AMT) FROM ORDERS);
